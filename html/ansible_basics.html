  domain: domain.com

variable precedence (least to most)

  1. role defaults
  2. inventory vars
  3. inventory group_vars
  4. inventory host_vars
  5. playbook group_vars
  6. playbook host_vars
  7. host facts
  8. registered vars

variable precedence (cont)

  1. set_facts
  2. play vars
  3. play vars_prompt
  4. play vars_files
  5. role and include vars
  6. block vars (only for tasks in block)
  7. task vars (only for the task)
  8. extra vars (always win precedence)

reboot


- name: reboot
  command: /usr/sbin/shutdown -r +1 "rebooting in 1 min" 
  async: 0
  poll: 0
  sudo: true
  ignore_errors: true
- name: wait for server to come back up
  delegate_to: localhost
  wait_for: host=\{{ inventory_hostname }}
            state=started
            port=22
            delay=120
            timeout=300
            connect_timeout=15
